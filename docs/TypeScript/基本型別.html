let obj = {
  [sym]: "semlinker",
};

Array​

let list: number[] = [1, 2, 3];

Enum​

一般使用​

enum Direction {
  NORTH, // 編譯後為 0
  SOUTH, // 1
  EAST, // 2
  WEST, // 3
}

let dir: Direction = Direction.NORTH; // 0

有初始的得使用​

enum Direction {
  NORTH = 3, // 編譯後為 3
  SOUTH, // 4
  EAST, // 5
  WEST, // 6
}

let dir: Direction = Direction.NORTH; // 3

字串的使用​

enum Direction {
  NORTH = 'NORTH', // NORTH
  SOUTH = 'SOUTH', // SOUTH
  EAST = 'EAST', // EAST
  WEST = 'WEST', // WEST
}

let dir: Direction = Direction.NORTH; // NORTH

常量枚舉​

在 enum 前面有加 const ，差別在於是否會全部都編譯成 js：Example，可以看 js 編譯後是否有 const 的差別

const enum Direction {
  NORTH, // 編譯後為 0
  SOUTH, // 1
  EAST, // 2
  WEST, // 3
}

let dir: Direction = Direction.NORTH; // 0

混合使用​

enum Enum {
  A,
  B,
  C = "C",
  D = "D",
  E = 8,
  F,
}

Any​

any 為代表，此參數可以為任何可能，無需檢查型別，但 any 太寬鬆了，如果要使用，可以先用 unknown

let notSure: any = 666;
notSure = "semlinker";
notSure = false;

Unknown​

any 與 unknown 都屬於，當不知道型別時，可以暫時賦予的型別 any 與 unknown 最大的差別在於

  1. 在賦值時，any 可以賦予給任何型別，但是 unknown 只能賦予給 any 或 unknown 型別的參數，
any
let value: any;

let value1: unknown = value; // OK
let value2: any = value; // OK
let value3: boolean = value; // OK
let value4: number = value; // OK
let value5: string = value; // OK
let value6: object = value; // OK
let value7: any[] = value; // OK
let value8: Function = value; // OK
unknown
let value: unknown;

let value1: unknown = value; // OK
let value2: any = value; // OK
let value3: boolean = value; // Error
let value4: number = value; // Error
let value5: string = value; // Error
let value6: object = value; // Error
let value7: any[] = value; // Error
let value8: Function = value; // Error
  1. any 在語法上為，任何型別都可以，unknown 語法上為，不知道型別，所以在操作這兩個型別的參數就會有差別
any
let value: any;

value.foo.bar; // OK
value.trim(); // OK
value(); // OK
new value(); // OK
value[0][1]; // OK
unknown
let value: unknown;

value.foo.bar; // Error
value.trim(); // Error
value(); // Error
new value(); // Error
value[0][1]; // Error

Tuple​

當一個變數固定欄位的地方需要固定的型別時，可以用 Tuple 規範

let tupleType: [string, boolean];
tupleType = ["semlinker", true];
tupleType = [true, "semlinker"]; // Error
tupleType = ["semlinker"]; // Error

Void​

宣告當 function 沒有回傳值，可以用 void

// 声明函数返回值为void
function warnUser(): void {
  console.log("This is my warning message");
}

let u: undefined = undefined;
let n: null = null;

Never​

Never 預設為永不存在的類型，當作是型別最後一層的把關

type Foo = string | number;

function controlFlowAnalysisWithNever(foo: Foo) {
  if (typeof foo === "string") {
    // foo 為 string
  } else if (typeof foo === "number") {
    // foo 為 number
  } else {
    // foo 為 never
    const check: never = foo;
  }
}

當有人改了 Foo 的型別，但是沒有改裡面的判斷時，就會有 never 被 assign 為其他型別，就會報錯，這時就可以避免改型別會用到的地方有問題

type Foo = string | number | boolean; // <- 新增了 boolean

function controlFlowAnalysisWithNever(foo: Foo) {
  if (typeof foo === "string") {
    // foo 為 string
  } else if (typeof foo === "number") {
    // foo 為 number
  } else {
    // 當 boolean assign 給 never 就會報錯
    const check: never = foo;
  }
}
